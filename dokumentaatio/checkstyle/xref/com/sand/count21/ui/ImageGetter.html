10  import javax.imageio.ImageIO;
11  
12  /**
13   * This class is a helper class for getting the card Images to the GUI.
14   * 
15   * @author osand
16   */
17  public class ImageGetter {
18  
19      /**
20       * This method gets the image of the card. 
21       * @param card card that needs a image
22       * @return the image of the card
23       */
24      public Image getCardImage(Card card) {
25          
26          String file = card.getFileName();
27          Image image = this.getImage(file);
28          return image;
29      }
30  
31      /**
32       * This method gets a image when it is given a filename.
33       * @param filename filename for the image wanted
34       * @return The image from the filename
35       */
36      public Image getImage(String filename) {
37          Image image = null;
38          InputStream is = this.getClass().getClassLoader().getResourceAsStream(filename);
39          try {
40              image = ImageIO.read(is);
41          } catch (IOException ex) {
42              Logger.getLogger(GUI.class.getName()).log(Level.SEVERE, null, ex);
43          }
44          image = image.getScaledInstance(250, 363, Image.SCALE_DEFAULT);
45          return image;
46      }
47  }
