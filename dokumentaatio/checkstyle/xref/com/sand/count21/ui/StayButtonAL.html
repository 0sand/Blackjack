9   import com.sand.count21.logiikka.Game;
10  import com.sand.count21.logiikka.ImageGetter;
11  import com.sand.count21.logiikka.Player;
12  import java.awt.Color;
13  import java.awt.Image;
14  import java.awt.event.ActionEvent;
15  import java.util.ArrayList;
16  import javax.swing.ImageIcon;
17  import javax.swing.JLabel;
18  import javax.swing.JPanel;
19  
20  /**
21   *
22   * @author osand
23   */
24  public class StayButtonAL implements java.awt.event.ActionListener {
25  
26      private Game game;
27      private ImageGetter imageGetter;
28      private JPanel dealerPanel;
29      private GUI gui;
30  
31      public StayButtonAL(Game game, ImageGetter imageGetter, GUI gui) {
32          this.gui = gui;
33          this.game = game;
34          this.imageGetter = imageGetter;
35          this.dealerPanel = gui.getDealerPanel();
36      }
37  
38  
39  
40      @Override
41      public void actionPerformed(ActionEvent e) {
42          if (game.getBetManager().getBet() > game.getPlayer().getMoney()) {
43              game.getBetManager().setBet(game.getPlayer().getMoney());
44          }
45  
46          dealerPanel.removeAll();
47          ArrayList<Card> cards = game.getDealer().getCards();
48          Card card1 = cards.get(0);
49          Card card2 = cards.get(1);
50          Image image1 = imageGetter.getCardImage(card1);
51          Image image2 = imageGetter.getCardImage(card2);
52  
53          JLabel dealerCard1 = new JLabel(new ImageIcon(image1));
54          JLabel dealerCard2 = new JLabel(new ImageIcon(image2));
55          dealerPanel.add(dealerCard1);
56          dealerPanel.add(dealerCard2);
57          dealerPanel.revalidate();
58          
59  
60          Player dealer = game.getDealer();
61  
62          while (game.checkIfDealerMustHit(dealer)) {
63  
64              game.dealOneTo(dealer);
65              Card card = dealer.getLastCard();
66              Image image = imageGetter.getCardImage(card);
67              JLabel dealerCardHit = new JLabel(new ImageIcon(image));
68              dealerPanel.add(dealerCardHit);
69  
70              dealerPanel.repaint();
71              dealerPanel.revalidate();
72  
73          }
74  
75          gui.getAgainButton().setEnabled(true);
76          gui.getHitButton().setEnabled(false);
77          gui.getStayButton().setEnabled(false);
78          gui.getBetButton().setEnabled(true);
79          gui.getDecreaseBetButton().setEnabled(true);
80  
81          //pays but do not zero bet
82          game.blackjackRoundDone();
83          if (game.didPlayerWin()) {
84              gui.getInfoField().setText("Player won");
85  
86          } else {
87              gui.getInfoField().setText("Player lost");
88          }
89          gui.getInfoField().setForeground(Color.red);
90  
91          int playerWins = game.getPlayer().getGamesWon();
92          gui.getNumberOfWins().setText("Player Wins  " + playerWins);
93          gui.getCurrentBetField().setText("bet " + (game.getBetManager().getBet()));
94          gui.getPlayerMoneyField().setText("Money " + game.getPlayer().getMoney());
95          gui.getCurrentBetField().repaint();
96          gui.getPlayerMoneyField().repaint();
97  
98      }
99  
100 }
