6   package com.sand.count21.ui;
7   
8   import com.sand.count21.logiikka.Card;
9   import com.sand.count21.logiikka.Game;
10  import com.sand.count21.logiikka.ImageGetter;
11  import com.sand.count21.logiikka.Player;
12  import java.awt.Image;
13  import java.awt.event.ActionEvent;
14  import java.awt.event.ActionListener;
15  import javax.swing.ImageIcon;
16  import javax.swing.JLabel;
17  
18  /**
19   *
20   * @author osand
21   */
22  public class HitButtonListener implements ActionListener{
23      private Game game;
24      private GUI gui;
25      private ImageGetter imageGetter;
26  
27      public HitButtonListener(Game game, GUI gui, ImageGetter imageGetter) {
28          this.game = game;
29          this.gui = gui;
30          this.imageGetter = imageGetter;
31      }
32  
33          @Override
34          public void actionPerformed(ActionEvent e) {
35              Player player = game.getPlayer();
36              if (!game.checkIfPlayerIsBust(player)) {
37                  game.dealTo(player, 1);
38                  Card card = player.getLastCard();
39                  Image image = imageGetter.getCardImage(card);
40                  JLabel playerCardHit = new JLabel(new ImageIcon(image));
41                  gui.getPlayerPanel().add(playerCardHit);
42                  gui.getPlayerPanel().revalidate();
43              }
44  
45              if (game.checkIfPlayerIsBust(player)) {
46  
47                  gui.getHitButton().setEnabled(false);
48                  for (ActionListener a : gui.getStayButton().getActionListeners()) {
49                      a.actionPerformed(new ActionEvent(this, ActionEvent.ACTION_PERFORMED, null) {
50                          //Nothing need go here, the actionPerformed method (with the
51                          //above arguments) will trigger the respective listener
52                      });
53                  }
54              }
55  
56          }
57      
58  }
