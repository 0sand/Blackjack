87      public class HitButton implements ActionListener {
88  
89          @Override
90          public void actionPerformed(ActionEvent e) {
91              Player player = game.getPlayer();
92              if (!game.checkIfPlayerIsBust(player)) {
93                  game.dealTo(player, 1);
94                  Card card = player.getLastCard();
95                  Image image = imageGetter.getCardImage(card);
96                  playerCardHit = new JLabel(new ImageIcon(image));
97                  playerPanel.add(playerCardHit);
98                  playerPanel.revalidate();
99              }
100 
101             if (game.checkIfPlayerIsBust(player)) {
102                 hitButton.setEnabled(false);
103             }
104 
105         }
106     }

108     public class AgainButton implements ActionListener {

110         @Override

112             againButton.setEnabled(false);

114             hitButton.setEnabled(true);
115             game.everyOneFolds();
116 
117             playerPanel.removeAll();

120             dealerPanel.removeAll();
121             dealerPanel.repaint();
122 
123             game.firstCardsInRound();
124             ArrayList<Card> cards = game.getPlayer().getCards();
125             Card card1 = cards.get(0);
126             Image image1 = imageGetter.getCardImage(card1);
127             Card card2 = cards.get(1);
128             Image image2 = imageGetter.getCardImage(card2);
129 
130             playerCard0 = new JLabel(new ImageIcon(image1));
131             playerCard1 = new JLabel(new ImageIcon(image2));
132             playerPanel.add(playerCard0);
133             playerPanel.add(playerCard1);
134 
135             cards = game.getDealer().getCards();
136             card1 = cards.get(0);
137             image1 = imageGetter.getImageFromString("images/backcard.png");
138             card2 = cards.get(1);
139             image2 = imageGetter.getCardImage(card2);
140 
141             dealerCard1 = new JLabel(new ImageIcon(image1));
142             dealerCard2 = new JLabel(new ImageIcon(image2));
143             dealerPanel.add(dealerCard1);
144             dealerPanel.add(dealerCard2);
145 
146             playerPanel.revalidate();
147             dealerPanel.revalidate();
148

166     class MyTimerActionListener implements ActionListener {
167 
168         @Override
169         public void actionPerformed(ActionEvent e) {
170 
171             dealerPanel.removeAll();
172             ArrayList<Card> cards = game.getDealer().getCards();
173             Card card1 = cards.get(0);
174             Card card2 = cards.get(1);     
175             Image image1 = imageGetter.getCardImage(card1);
176             Image image2 = imageGetter.getCardImage(card2);
177             
178             
179             dealerCard1 = new JLabel(new ImageIcon(image1));
180             dealerCard2 = new JLabel(new ImageIcon(image2));
181             dealerPanel.add(dealerCard1);
182             dealerPanel.add(dealerCard2);
183             dealerPanel.revalidate();
184 
185 
186             Player dealer = game.getDealer();
187 
188             while (game.checkIfDealerMustHit(dealer)) {
189 
190                 game.dealTo(dealer, 1);
191                 Card card = dealer.getLastCard();
192                 Image image = imageGetter.getCardImage(card);
193                 dealerCardHit = new JLabel(new ImageIcon(image));
194                 dealerPanel.add(dealerCardHit);
195                 dealerPanel.revalidate();
196 
197             }
198             againButton.setEnabled(true);
199             hitButton.setEnabled(false);
200             stayButton.setEnabled(false);
201             timer.stop();
202         }
